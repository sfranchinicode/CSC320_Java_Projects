BEGIN

    //Initialize variables
    Initialize total to 0
    Initialize count to 0
    Initialize max to 0
    Initialize min to 0

    // Use WHILE loop to control iterations
    WHILE count is less than 5 DO
        PRINT "Enter your value: "
        READ num

        // Use IF loop to update max and min as inputs are received
        IF count equals 0 THEN
            SET max to num
            SET min to num
        ELSE
            IF num is greater than max THEN
                SET max to num
            ENDIF
            IF num is less than min THEN
                SET min to num
            ENDIF
        ENDIF
        SET total equal to total + num
        INCREMENT count by 1
    END WHILE
    
    SET average to total divided by count
    SET interest to total multiplied by 0.20
    
    // Display results
    PRINT "The total of the numbers is: " + total
    PRINT "The average of the numbers is: " + average
    PRINT "The maximum number input was: " + max
    PRINT "The minimum number input was: " + min
    PRINT "The interest on the total is: " + interest
END
